#include "STC8H.h"
#include <string.h>

// 从机I2C地址（7位）
#define I2C_SLAVE_ADDR  0x60

// 命令码定义（与主机一致）
#define CMD_START_DISCHARGE     0x01
#define CMD_STOP_DISCHARGE      0x02
#define CMD_START_CHARGE        0x03
#define CMD_STOP_CHARGE         0x04
#define CMD_ALARM_OVERCHARGE    0x05
#define CMD_ALARM_OVERDISCHARGE 0x06
#define CMD_SET_OVERC_PROT_VOLT 0x07
#define CMD_SET_OVERD_PROT_VOLT 0x08
#define CMD_CUR_CHARGE_MAH      0x09
#define CMD_CUR_DISCHARGE_MAH   0x0A
#define CMD_CUR_BAT_VOLT        0x0B
#define CMD_READ_OVERC_PROT_VOLT 0x0C
#define CMD_READ_OVERD_PROT_VOLT 0x0D
#define CMD_SET_CYCLE_DISCH_NUM 0x0E
#define CMD_CUR_CYCLE_NUM       0x0F
#define CMD_CYCLE_DISCH_END_MAH_1  0x10
// 后续依次到0x19
#define CMD_CYCLE_CHARGE_END_MAH_1 0x1A
// 后续依次到0x23

// ---------- 内部变量 ----------
// 所有数据存储在xdata区（根据需要可调整）
xdata uint16_t curChargeMAH;            // 当前充电mAh
xdata uint16_t curDischargeMAH;         // 当前放电mAh
xdata uint16_t curBatVolt;              // 当前电池电压(mV)
xdata uint16_t overchargeProtectVolt;   // 设置的过充保护电压
xdata uint16_t overdischargeProtectVolt;// 设置的过放保护电压
xdata uint16_t readOverchargeProtectVolt;  // 实测过充保护电压
xdata uint16_t readOverdischargeProtectVolt; // 实测过放保护电压
xdata uint8_t cycleDischargeNum;        // 写入的循环放电次数
xdata uint8_t curCycleNum;              // 当前循环次数
xdata uint16_t cycleDischargeEndMAH[10]; // 10次放电结束mAh
xdata uint16_t cycleChargeEndMAH[10];   // 10次充电结束mAh
xdata uint8_t alarmOvercharge;          // 过充报警位 (bit0)
xdata uint8_t alarmOverdischarge;       // 过放报警位 (bit0)

// ---------- I2C从机状态控制 ----------
xdata uint8_t i2c_cmd;                  // 当前接收的命令
xdata uint8_t i2c_rx_len;               // 需要接收的数据长度（0/1/2）
xdata uint8_t i2c_rx_index;              // 已接收字节数
xdata uint8_t i2c_rx_buf[2];             // 接收数据缓冲区
xdata uint8_t i2c_tx_len;                // 待发送数据长度
xdata uint8_t i2c_tx_index;              // 已发送字节数
xdata uint8_t i2c_tx_buf[2];             // 发送缓冲区
xdata bit i2c_pending_read;              // 等待读请求标志

// 函数声明
void I2C_Init(void);
void I2C_Isr(void) interrupt 24;  // STC8H I2C中断号一般为24，请参考具体型号

// ---------- 初始化I2C从机 ----------
void I2C_Init(void) {
    // 配置I2C从机地址，使能中断
    I2CSLADR = I2C_SLAVE_ADDR;   // 设置7位地址（硬件自动处理读写位）
    I2CCFG = 0x01;                // 使能I2C从机模式
    I2CMSST = 0x00;
    I2CSLST = 0x00;
    I2CIF = 0;                    // 清中断标志
    EA = 1;                       // 开总中断
}

// ---------- I2C中断服务程序 ----------
void I2C_Isr(void) interrupt 24 {
    if (I2CSLST & 0x40) { // 检测到START信号
        I2CSLST &= ~0x40;  // 清除START标志
        i2c_rx_index = 0;
        i2c_tx_index = 0;
        i2c_pending_read = 0;
    }

    if (I2CSLST & 0x20) { // 接收到地址且匹配
        if (I2CSLST & 0x10) { // 是读请求（主机读从机）
            // 根据i2c_cmd准备发送数据
            switch (i2c_cmd) {
                case CMD_ALARM_OVERCHARGE:
                    i2c_tx_buf[0] = alarmOvercharge;
                    i2c_tx_len = 1;
                    break;
                case CMD_ALARM_OVERDISCHARGE:
                    i2c_tx_buf[0] = alarmOverdischarge;
                    i2c_tx_len = 1;
                    break;
                case CMD_CUR_CHARGE_MAH:
                    i2c_tx_buf[0] = curChargeMAH >> 8;
                    i2c_tx_buf[1] = curChargeMAH & 0xFF;
                    i2c_tx_len = 2;
                    break;
                case CMD_CUR_DISCHARGE_MAH:
                    i2c_tx_buf[0] = curDischargeMAH >> 8;
                    i2c_tx_buf[1] = curDischargeMAH & 0xFF;
                    i2c_tx_len = 2;
                    break;
                case CMD_CUR_BAT_VOLT:
                    i2c_tx_buf[0] = curBatVolt >> 8;
                    i2c_tx_buf[1] = curBatVolt & 0xFF;
                    i2c_tx_len = 2;
                    break;
                case CMD_READ_OVERC_PROT_VOLT:
                    i2c_tx_buf[0] = readOverchargeProtectVolt >> 8;
                    i2c_tx_buf[1] = readOverchargeProtectVolt & 0xFF;
                    i2c_tx_len = 2;
                    break;
                case CMD_READ_OVERD_PROT_VOLT:
                    i2c_tx_buf[0] = readOverdischargeProtectVolt >> 8;
                    i2c_tx_buf[1] = readOverdischargeProtectVolt & 0xFF;
                    i2c_tx_len = 2;
                    break;
                case CMD_CUR_CYCLE_NUM:
                    i2c_tx_buf[0] = curCycleNum;
                    i2c_tx_len = 1;
                    break;
                // 循环放电结束mAh (0x10~0x19)
                case CMD_CYCLE_DISCH_END_MAH_1 ... CMD_CYCLE_DISCH_END_MAH_1 + 9:
                {
                    uint8_t idx = i2c_cmd - CMD_CYCLE_DISCH_END_MAH_1;
                    uint16_t val = cycleDischargeEndMAH[idx];
                    i2c_tx_buf[0] = val >> 8;
                    i2c_tx_buf[1] = val & 0xFF;
                    i2c_tx_len = 2;
                }
                break;
                // 循环充电结束mAh (0x1A~0x23)
                case CMD_CYCLE_CHARGE_END_MAH_1 ... CMD_CYCLE_CHARGE_END_MAH_1 + 9:
                {
                    uint8_t idx = i2c_cmd - CMD_CYCLE_CHARGE_END_MAH_1;
                    uint16_t val = cycleChargeEndMAH[idx];
                    i2c_tx_buf[0] = val >> 8;
                    i2c_tx_buf[1] = val & 0xFF;
                    i2c_tx_len = 2;
                }
                break;
                default:
                    i2c_tx_len = 0; // 无数据
                    break;
            }
            i2c_tx_index = 0;
            // 将第一个字节写入发送寄存器
            if (i2c_tx_len > 0) {
                I2CTXD = i2c_tx_buf[0];
                i2c_tx_index = 1;
            }
        } else { // 是写请求（主机写从机）
            // 准备接收第一个字节（命令码）
            i2c_rx_len = 1; // 先接收命令码
        }
        I2CSLST &= ~0x30; // 清除地址标志和读写标志位
    }

    if (I2CSLST & 0x08) { // 接收到数据字节（主机写数据）
        uint8_t dat = I2CRXD;
        if (i2c_rx_index == 0) {
            // 第一个字节是命令码
            i2c_cmd = dat;
            // 根据命令码确定后续数据长度
            switch (i2c_cmd) {
                case CMD_SET_OVERC_PROT_VOLT:
                case CMD_SET_OVERD_PROT_VOLT:
                    i2c_rx_len = 2; // 需要2字节数据
                    break;
                case CMD_SET_CYCLE_DISCH_NUM:
                    i2c_rx_len = 1;
                    break;
                default:
                    // 无数据或读命令，后续无需接收
                    i2c_rx_len = 0;
                    break;
            }
            i2c_rx_index = 1; // 已接收命令码
        } else {
            // 接收数据字节
            if (i2c_rx_index - 1 < sizeof(i2c_rx_buf)) {
                i2c_rx_buf[i2c_rx_index - 1] = dat;
            }
            i2c_rx_index++;
        }
        I2CSLST &= ~0x08; // 清除接收标志
    }

    if (I2CSLST & 0x04) { // 需要发送数据（主机读数据）
        if (i2c_tx_index < i2c_tx_len) {
            I2CTXD = i2c_tx_buf[i2c_tx_index];
            i2c_tx_index++;
        } else {
            I2CTXD = 0xFF; // 发送默认值
        }
        I2CSLST &= ~0x04; // 清除发送标志
    }

    if (I2CSLST & 0x01) { // 检测到STOP信号
        // 执行写命令的实际操作
        if (i2c_rx_len > 0) {
            // 有数据接收完成，根据命令更新变量
            switch (i2c_cmd) {
                case CMD_SET_OVERC_PROT_VOLT:
                    overchargeProtectVolt = (i2c_rx_buf[0] << 8) | i2c_rx_buf[1];
                    // 可触发硬件动作
                    break;
                case CMD_SET_OVERD_PROT_VOLT:
                    overdischargeProtectVolt = (i2c_rx_buf[0] << 8) | i2c_rx_buf[1];
                    break;
                case CMD_SET_CYCLE_DISCH_NUM:
                    cycleDischargeNum = i2c_rx_buf[0];
                    // 可重置循环计数器等
                    break;
                default:
                    break;
            }
        } else {
            // 无数据写命令，直接执行动作
            switch (i2c_cmd) {
                case CMD_START_DISCHARGE:
                    // 启动放电（例如置位IO）
                    P10 = 1; // 示例：P1.0控制放电MOS
                    break;
                case CMD_STOP_DISCHARGE:
                    P10 = 0;
                    break;
                case CMD_START_CHARGE:
                    P11 = 1;
                    break;
                case CMD_STOP_CHARGE:
                    P11 = 0;
                    break;
                default:
                    break;
            }
        }
        // 清空状态
        i2c_rx_index = 0;
        i2c_tx_index = 0;
        I2CSLST &= ~0x01; // 清除STOP标志
    }
}

// ---------- 主函数 ----------
void main(void) {
    // 初始化IO（例如控制充放电的引脚）
    P1M0 = 0x03; // P1.0 P1.1 推挽输出
    P1M1 = 0x00;
    P10 = 0;
    P11 = 0;

    // 初始化变量
    curChargeMAH = 0;
    curDischargeMAH = 0;
    curBatVolt = 0;
    overchargeProtectVolt = 4200; // 默认4.2V
    overdischargeProtectVolt = 3000;
    readOverchargeProtectVolt = 0;
    readOverdischargeProtectVolt = 0;
    cycleDischargeNum = 0;
    curCycleNum = 0;
    alarmOvercharge = 0;
    alarmOverdischarge = 0;
    memset(cycleDischargeEndMAH, 0, sizeof(cycleDischargeEndMAH));
    memset(cycleChargeEndMAH, 0, sizeof(cycleChargeEndMAH));

    // 初始化I2C
    I2C_Init();

    while (1) {
        // 主循环可执行其他任务，如采集电压、更新变量等
        // 此处示例：简单延时并更新模拟值
        // 实际应用中，可根据硬件定时更新 curBatVolt, curChargeMAH 等
    }
}
